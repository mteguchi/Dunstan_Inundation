# JAGS model to fit a linear model to water height vs. distance from berm


model{
	for (i in 1:N){
		mu.y[i] <- b0[transect[i]] + b1[transect[i]] * Dist_Berm[i] + b2[year[i]] * year[i] + b3[month[i]] * month[i]
		y[i] ~ dnorm(mu.y[i], tau.y)  
	}
	sigma.y <- sqrt(1/tau.y)

	for (k in 1:4){
		b0[k] ~ dnorm(0, 0.01)
		b1[k] ~ dnorm(0, 0.01)
	}

	for (k in 1:2){
		b2[k] ~ dnorm(0, 0.01)
	}

	for (k in 1:7){
		b3[k] ~ dnorm(0, 0.01)
	}

}
