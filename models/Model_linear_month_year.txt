# JAGS model to fit a linear model to water height vs. distance from berm


model{
	for (i in 1:N){
		mu.y[i] <- b0[month[i]] + b1[month[i]] * Dist_Berm[i] + b2[year[i]] * year[i]
		y[i] ~ dnorm(mu.y[i], tau.y)  
	}
	
	sigma.y <- sqrt(1/tau.y)

	tau.y ~ dgamma(0.1, 0.01)

	for (k in 1:n.months){
		b0[k] ~ dnorm(0, 0.01)
		b1[k] ~ dnorm(0, 0.01)
	}

	for (k in 1:n.years){
		b2[k] ~ dnorm(0, 0.01)
	}

}
